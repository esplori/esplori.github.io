在网上找了一份代码可以用来对css进行格式化，主要是去除空格，也可以用来格式化js

            <!DOCTYPE html>
            <html lang="en">
            <head>
                <meta charset="UTF-8">
                <title>css压缩</title>
            </head>
            <body >

            <div style="width: 60%;margin: 5% auto">
                <textarea type="text" id="receiveInput"  rows="20"  style="width: 100%;"></textarea>
                <button id="decodeBtn" onclick="getCssCode(this)">格式化</button>
                <button id="encodeBtn" onclick="getCssCode(this)">压缩</button>
                <textarea type="text" id="changedInput"  rows="20"  style="width: 100%;"></textarea>
            </div>
            <script>

                function getCssCode(item){
                    var inputCode=document.getElementById("receiveInput").value,outpunCode=document.getElementById("changedInput"),ecode=null;
                    if(item.id=='decodeBtn'){
                        ecode=CSSdecode(inputCode);
                    }else{
                        ecode=CSSencode(inputCode);
                    }
                    outpunCode.value=ecode;
                }
                // 压缩
                function CSSencode(code) {
                        code = code.replace(/\r\n/ig,'');
                        code = code.replace(/(\s){2,}/ig,'$1');
                        code = code.replace(/\t/ig,'');
                        code = code.replace(/\n\}/ig,'\}');
                        code = code.replace(/\n\{\s*/ig,'\{');
                        code = code.replace(/(\S)\s*\}/ig,'$1\}');
                        code = code.replace(/(\S)\s*\{/ig,'$1\{');
                        code = code.replace(/\{\s*(\S)/ig,'\{$1');
                        return code;
                    }
                    //格式化
                function CSSdecode(code) {
                    code = code.replace(/(\s){2,}/ig,'$1');
                    code = code.replace(/(\S)\s*\{/ig,'$1 {');
                    code = code.replace(/\*\/(.[^\}\{]*)}/ig,'\*\/\n$1}');
                    code = code.replace(/\/\*/ig,'\n\/\*');
                    code = code.replace(/;\s*(\S)/ig,';\n\t$1');
                    code = code.replace(/\}\s*(\S)/ig,'\}\n$1');
                    code = code.replace(/\n\s*\}/ig,'\n\}');
                    code = code.replace(/\{\s*(\S)/ig,'\{\n\t$1');
                    code = code.replace(/(\S)\s*\*\//ig,'$1\*\/');
                    code = code.replace(/\*\/\s*([^\}\{]\S)/ig,'\*\/\n\t$1');
                    code = code.replace(/(\S)\}/ig,'$1\n\}');
                    code = code.replace(/(\n){2,}/ig,'\n');
                    code = code.replace(/:/ig,':');
                    code = code.replace(/  /ig,' ');
                    return code;
                }


            </script>

            </body>
            </html>
