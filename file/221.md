function pushDate(array,date) {
    array[0]=date.getFullYear();array[1]=date.getMonth()+1;array[2]=date.getDate();
    array[1]=array[1]&lt;10?"0"+array[1]:array[1]
    array[2]=array[2]&lt;10?"0"+array[2]:array[2]
    return array；
}
var t=[],w=[],n=new Date();
t=pushDate(t, n);
var m= new Date(t[0],t[1],0).getDate();//获取当月天数
for(var i=1;i&lt;m;i++){
    var d= new Date(t[0],t[1]-1,i);
    var day=d.getDay();//判断星期天、星期六
    if(day!=6&amp;&amp;day!=0){
        pushDate(t,d);
        w.push(t.join("-"))
    }
}
console.log(w)
