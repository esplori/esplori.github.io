h5的localStorage在实际应用中很方便，不同页面可以很方便共享数据，所以 用下面这种全局方法可以方便对数据进行设置、清空以及取值。
<pre class="prettyprint linenums">      var handleData={
        data:function (key,value) {

            if(value==''){
                localStorage.removeItem(key);
            }else if(value==undefined){
                return localStorage.getItem(key);
            }else{
                return   localStorage.setItem(key,value);
            }

        }
    }

    var data=handleData.data("test")//从localStorage取值
    var data=handleData.data("test",'')//从localStorage清空
   var data=handleData.data("test",'hello world')//设置localStorage
</pre>
